Función edit(array())

Esta función edita un clasificado existente a través de parámetros recibidos a través de un arreglo (array)

La función recibe cualquiera de los siguientes parámetos y deben estar presentes en el areglo sin importar que se encuentren vacíos

El id del clasificado es el valor mas importante pues indica cuál es el registro que se editará y único por ser de tipo llave
- idclasificado (numérico, llave)
- titulo (alfanumérico)
- precio (numérico flotante)
- preciodiv (numérico 1 caracter)
- detalles (alfanumérico)
- marca (alfanumérico)
- modelo (alfanumérico)
- estado (alfanumérico)

<?php

// Asigna datos para crear un nuevo clasificado 
$new_clasificado_data = array(
    'titulo'=>'Mustang', 'precio'=>'36000', 'preciodiv'=>'1', 'detalles'=>'Excelente auto restaurado y en perfectas condiciones Mustang 64 Cobra todo un clasico', 'marca'=>'Ford', 'modelo'=>'Mustang', 'estado'=>'1964', 'idusuario'=>'3', 'nombreimagen'=>'mustang64.jpg'
);
// Genera el objeto para crear el clasificado
$clasificado3 = new Clasificado();
// Crea el clasificado con los datos contenidos en $new_clasificado_data
$clasificado3->set($new_clasificado_data);

// Edita del clasificado agregado anteriormente
// En este caso sabemos que su ID es 2
// Para editar debemos conocer el ID que se quiere editar
$edit_clasificado_data = array(
    'idclasificado'=>'2', 'titulo'=>'Mustang Cobra 64 Clásico', 'precio'=>'30000', 'preciodiv'=>'0', 'detalles'=>'Excelente opción restaurado y en perfectas condiciones Mustang 64 Cobra todo un clasico', 'marca'=>'Ford', 'modelo'=>'Mustang Cobra', 'estado'=>'1964'
);
// Genera el objeto para editar el clasificado
$clasificado4 = new Clasificado();
// Edita el clasificado con los datos contenidos en $edit_clasificado_data
$clasificado4->edit($edit_clasificado_data);

// Obtenemos los datos que se editaron para ver los cambios
$clasificado5 = new Clasificado();
$clasificado5->get(2);

?>
